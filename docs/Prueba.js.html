/**

 *
 * @param {string} ruta La ruta de la base de datos que queremos obtener
 * @return {*} 
 */
function get(ruta) {
    const dbRef = ref(getDatabase());
    return get(child(dbRef, url))
        .then((snapshot) => {
            if (snapshot.exists()) {
                console.log(snapshot.val());
                return snapshot.val();
            } else {
                console.log("No data available");
                return false;
            }
        });
}
